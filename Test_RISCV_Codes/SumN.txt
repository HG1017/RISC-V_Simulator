addi x3,x0,10 #x3 = N
lui x18,0x10000
addi x18,x18,0x000 #x18 ->base address of array arr
addi x31,x0,2 #x31=2
addi x5,x0,0 #x5=i=0
addi x7,x0,0 #j=0
addi x8,x0,0 #sum=0
InitializeLoop:
	bge x5,x3,Sum
    sll x6,x5,x31 #x6 = 4*i
    add x6,x6,x18 #x6 = x18+4i ->address of arr[i]
    sw x5,0(x6) #arr[i] = i
    addi x5,x5,1
    blt x5,x3,InitializeLoop
Sum:
	bge x7,x3,exit
	sll x6,x7,x31 #x6 =4j
    add x6,x6,x18 #x6 -> address of arr[j]
    lw x9,0(x6) #x9 = arr[j]
    add x8,x8,x9 #sum+=arr[j]
    addi x7,x7,1
    blt x7,x3,Sum
exit:
